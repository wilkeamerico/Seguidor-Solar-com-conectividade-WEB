      const x = (index / (data.length - 1 || 1)) * w;

      const y = h - (norm * (h - 20)) - 10; // Margem de segurança de 10px
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    });
    return 'M ' + points.join(' L ');
  }

//  LÓGICA AUXILIAR / SIMULADOR ELÉTRICO
// =========================================================================
void simularEletrico(float dtS) {
  float al   = constrain(1.0f - (abs(angulo - 90) / 25.0f), 0.0f, 1.0f);
  corrente   = max(0.0f, I_PICO * al + (random(-30, 30) / 1000.0f));
  tensao     = V_BASE + (al * 0.2f) + (random(-5, 5) / 100.0f);
  potencia   = tensao * corrente;
  energiaWh += potencia * (dtS / 3600.0f);
}

  display.setTextSize(1);

  switch (tela) {
    case 0: // Rastreamento LDR
      display.setCursor(0, 0);  display.println("== RASTREAMENTO LDR ==");
      display.setCursor(0, 15); display.printf("LDR Esq: %d\n", ldrEsq);
      display.setCursor(0, 27); display.printf("LDR Dir: %d\n", ldrDir);
      display.setCursor(0, 39); display.printf("Dif:     %d (Tol:%d)\n", difLdr, TOLERAN);
      display.setCursor(0, 52); display.printf("Ang:     %d deg\n", angulo);
      break;

    case 1: // Elétrico Simulado
      display.setCursor(0, 0);  display.println("== METRICAS DA PLACA =");
      display.setCursor(0, 15); display.printf("Corr: %.3f A\n", corrente);
      display.setCursor(0, 27); display.printf("Tens: %.2f V\n", tensao);
      display.setCursor(0, 39); display.printf("Pot : %.3f W\n", potencia);
      display.setCursor(0, 52); display.printf("Ener: %.5f Wh\n", energiaWh);
      break;

    case 2: // Rede / AP Info
      display.setCursor(0, 0);  display.println("== DADOS DA CONEXAO ==");
      display.setCursor(0, 15); display.printf("SSID: %s\n", AP_SSID);
      display.setCursor(0, 27); display.print("IP  : "); display.println(WiFi.softAPIP());
      display.setCursor(0, 39); display.println("Acesse: 192.168.4.1");
      display.setCursor(0, 52); display.printf("Clients: %d\n", WiFi.softAPgetStationNum());
      break;
  }
  display.display();
}

// =========================================================================
//  SETUP
// =========================================================================
void setup() {
  Serial.begin(115200);
  delay(1000);

  Wire.begin();
  if (!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR)) {
    Serial.println("[ERRO] Display OLED nao iniciado!");
    while (1);
  }
  
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);
  display.setCursor(0, 0);
  display.println("Seguidor Solar V7.3");
  display.println("Iniciando AP...");
  display.display();

  pinMode(PIN_LDR_ESQ, INPUT);
  pinMode(PIN_LDR_DIR, INPUT);

  servo.setPeriodHertz(50);
  servo.attach(PIN_SERVO, 500, 2500);
  servo.writeMicroseconds(map(angulo, 0, 180, 500, 2500));

  WiFi.mode(WIFI_AP);
  WiFi.softAPConfig(AP_IP, AP_IP, IPAddress(255, 255, 255, 0));
  WiFi.softAP(AP_SSID, AP_PASS);

  server.on("/", handleRoot);
  server.on("/data", handleData);
  server.begin();

  randomSeed(esp_random());
  tLoop = millis();
}

// =========================================================================
//  LOOP DE CONTROLE PRINCIPAL
// =========================================================================
void loop() {
  server.handleClient();
  unsigned long agora = millis();

  // Ciclo Principal de Controle (300 ms)
  if (agora - tLoop >= T_LOOP) {
    float dtS = (agora - tLoop) / 1000.0f;
    tLoop     = agora;

    ldrEsq = analogRead(PIN_LDR_ESQ);
    ldrDir = analogRead(PIN_LDR_DIR);
    difLdr = ldrEsq - ldrDir;

    // Ajuste da posição do servo motor com base nos LDRs usando Operador Ternário
    posServo += (abs(difLdr) <= TOLERAN) ? 0.0f : ((difLdr > 0) ? -1.0f : 1.0f);
    posServo = constrain(posServo, ANG_MIN, ANG_MAX);
    angulo   = (int)posServo;
    servo.writeMicroseconds(map(angulo, 0, 180, 500, 2500));

    // Determina o status da operação via Operador Ternário
    statusTrk = (abs(difLdr) <= TOLERAN) ? "Alinhado" : ((difLdr > 0) ? "Mov. Horario" : "Mov. Anti-Horario");

    // Processa a simulação elétrica correspondente
    simularEletrico(dtS);

    // Sincronização e Envio das Informações pelo Monitor Serial
    Serial.println("\n=== MONITORAMENTO SEGUIDOR SOLAR ===");
    Serial.printf("Alinhamento : LDR_Esq: %4d | LDR_Dir: %4d | Diferenca: %4d\n", ldrEsq, ldrDir, difLdr);
    Serial.printf("Servo Motor : Angulo : %3d deg | Status   : %s\n", angulo, statusTrk);
    Serial.printf("Eletrico    : %.2f V  | %.3f A   | %.3f W\n", tensao, corrente, potencia);
    Serial.printf("Acumulado   : %.5f Wh\n", energiaWh);
    Serial.println("=====================================");
  }

  // Ciclo de Transição de Telas do OLED (3 segundos)
  if (agora - tOled >= T_OLED) {
    tOled = agora;
    tela  = (tela + 1) % 3;
    atualizarOLED();
  }
}
