// ---- Utilitários ----
function fmtNum(v, dec){ return (typeof v==='number'&&!isNaN(v))?v.toFixed(dec):'—'; }
function fmtTime(ms){
  const d=new Date(); // usamos o relógio do browser
  // ms é o uptime; calculamos o horário subtraindo o delta
  return d.toLocaleString('pt-BR');
}
function nowStr(){
  return new Date().toLocaleString('pt-BR',{hour12:false});
}

// ---- Gauge servo ----
function updateGauge(angle){
  // ângulo entre 65 e 115
  const pct = (angle - 65) / 50; // 0..1
  const arcLen = pct * 157; // comprimento total do semi-arco ~= πr = π*50
  document.getElementById('arcFill').setAttribute('stroke-dasharray', arcLen + ' 157');
  // agulha: mapear 65->-90deg, 90->0deg, 115->90deg
  const deg = (angle - 90) * (90/25);
  document.getElementById('needle').style.transform = `rotate(${deg}deg)`;
}

// ---- Tabela histórico ----
let dadosCSV = [];

function addLinhaTabela(d){
  dadosCSV.push(d);
  const tb = document.getElementById('tabelaBody');
  const tr = document.createElement('tr');
  tr.innerHTML = `
    <td>${d.hora}</td>
    <td>${fmtNum(d.corrente,3)}</td>
    <td>${fmtNum(d.tensao,2)}</td>
    <td>${fmtNum(d.potencia,3)}</td>
    <td>${fmtNum(d.energiaWh,4)}</td>
    <td>${fmtNum(d.energiaKWh,6)}</td>
    <td>${d.angulo}</td>
  `;
  // insere no topo
  if(tb.firstChild) tb.insertBefore(tr, tb.firstChild);
  else tb.appendChild(tr);
  // limita exibição a 200 linhas
  while(tb.rows.length > 200) tb.deleteRow(tb.rows.length-1);
  document.getElementById('totalAmostras').textContent = dadosCSV.length;
}

// ---- Exportar CSV ----
function exportarCSV(){
  if(dadosCSV.length===0){ alert('Nenhum dado coletado ainda.'); return; }
  const cab = 'Data/Hora,Corrente (A),Tensao (V),Potencia (W),Energia (Wh),Energia (kWh),Angulo (graus)\n';
  const linhas = dadosCSV.map(d =>
    `"${d.hora}",${fmtNum(d.corrente,4)},${fmtNum(d.tensao,2)},${fmtNum(d.potencia,4)},${fmtNum(d.energiaWh,5)},${fmtNum(d.energiaKWh,7)},${d.angulo}`
  ).join('\n');
  const blob = new Blob(['\uFEFF'+cab+linhas], {type:'text/csv;charset=utf-8'});
  const url  = URL.createObjectURL(blob);
  const a    = document.createElement('a');
  const ts   = new Date().toISOString().replace(/[:.]/g,'-').slice(0,19);
  a.href = url;
  a.download = `seguidor_solar_${ts}.csv`;
  a.click();
  URL.revokeObjectURL(url);
}

/* ======================================================== */
/*  FUNÇÃO AUXILIAR – LEITURA ACS712                        */
/* ======================================================== */
float lerCorrenteACS712(void) {
  long somaADC = 0;
  for (int i = 0; i < ACS712_AMOSTRAS; i++) {
    somaADC += analogRead(PIN_ACS712);
    delayMicroseconds(200);
  }
  float mediaADC   = (float)somaADC / ACS712_AMOSTRAS;
  float tensaoVout = mediaADC * (ADC_VREF / ADC_RESOLUCAO);
  float corrente   = (tensaoVout - ACS712_OFFSET_V) / ACS712_SENS_VA;
  return fabsf(corrente);
}

/* ======================================================== */
/*  SETUP                                                   */
/* ======================================================== */
void setup(void) {
  Serial.begin(115200);
  Serial.println("\n=== Rastreador Solar ESP32 V3 – Dashboard Web ===");

  /* ---- LCD 1602A ---- */
  lcd.begin(16, 2);
  lcd.clear();
  lcd.setCursor(0, 0); lcd.print(" Rastr. Solar V3");
  lcd.setCursor(0, 1); lcd.print(" Iniciando Wi-Fi");

  /* ---- Pinos Sensores ---- */
  pinMode(PIN_LDR_ESQ, INPUT);
  pinMode(PIN_LDR_DIR, INPUT);
  pinMode(PIN_ACS712,  INPUT);

  /* ---- Servo ---- */
  servoMotor.setPeriodHertz(50);
  servoMotor.attach(PIN_SERVO, 500, 2500);
  servoMotor.writeMicroseconds(1500);

  /* ---- Wi-Fi SoftAP ---- */
  WiFi.softAP(ssid, password);
  Serial.println("\n[OK] Rede Wi-Fi criada!");
  Serial.print("  SSID   : "); Serial.println(ssid);
  Serial.print("  Senha  : "); Serial.println(password);
  Serial.print("  URL    : http://"); Serial.println(WiFi.softAPIP());

  /* ---- WebServer — rotas ---- */
  server.on("/",     handleRoot);
  server.on("/data", handleData);   // <-- novo endpoint JSON
  server.begin();
  Serial.println("[OK] Servidor iniciado!");

  delay(2000);
  lcd.clear();
  tempoAnterior = millis();
}

/* ======================================================== */
/*  LOOP PRINCIPAL                                          */
/* ======================================================== */
void loop(void) {
  /* 1. Mantém o servidor Web responsivo */
  server.handleClient();
  unsigned long agora = millis();

  /* 2. Ciclo de leitura e controle do Servo */
  if (agora - tempoAnterior >= intervalo_loop) {
    float dtSegundos = (agora - tempoAnterior) / 1000.0f;
    tempoAnterior = agora;

    /* ---- LDRs ---- */
    valorLdrEsq = analogRead(PIN_LDR_ESQ);
    valorLdrDir = analogRead(PIN_LDR_DIR);
    diferenca   = valorLdrEsq - valorLdrDir;

    /* ---- Lógica de Tracker Incremental ---- */
    if (abs(diferenca) <= TOLERANCIA) {
      statusAcao = "Alinhado / Mantendo";
    } else {
      if (diferenca > 0) {
        posServo -= 1.0;
        statusAcao = "Ajustando anti-horário";
      } else {
        posServo += 1.0;
        statusAcao = "Ajustando horário";
      }
    }

    posServo  = constrain(posServo, ANGULO_MIN, ANGULO_MAX);
    pulsoServo = map((int)posServo, 0, 180, 500, 2500);
    servoMotor.writeMicroseconds(pulsoServo);

    /* ---- Corrente e Energia ---- */
    correnteA  = lerCorrenteACS712();
    potenciaW  = TENSAO_PLACA_V * correnteA;
    energiaWh  += potenciaW * (dtSegundos / 3600.0f);
    energiaKWh = energiaWh / 1000.0f;

    /* ---- Debug Serial ---- */
    Serial.printf(
      "LDR_E:%4d | LDR_D:%4d | Dif:%5d | Ang:%3d° | "
      "I:%.3fA | P:%.3fW | E:%.5fkWh\n",
      valorLdrEsq, valorLdrDir, diferenca, (int)posServo,
      correnteA, potenciaW, energiaKWh
    );
  }

  /* 3. Atualização do LCD 1602A */
  if (agora - ultimoLCD >= INTERVALO_LCD_MS) {
    ultimoLCD = agora;
    telaLCD   = (telaLCD + 1) % 2;
    lcd.clear();

    switch (telaLCD) {
      case 0:
        lcd.setCursor(0, 0);
        lcd.print("Inclin:");
        if ((int)posServo < 100) lcd.print(" ");
        lcd.print((int)posServo);
        lcd.print((char)0xDF);
        lcd.setCursor(0, 1);
        lcd.print("Corr: ");
        lcd.print(correnteA, 2);
        lcd.print("A       ");
        break;
      case 1:
        lcd.setCursor(0, 0);
        lcd.print("Pot:");
        lcd.print(potenciaW, 2);
        lcd.print("W       ");
        lcd.setCursor(0, 1);
        lcd.print("En:");
        lcd.print(energiaKWh, 5);
        lcd.print("kWh ");
        break;
    }
  }
}
